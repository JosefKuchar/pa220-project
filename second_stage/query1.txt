# Query 1

SELECT Count(*),
       Date(time_conn),
       program_ver
FROM   staging.import_connection
WHERE  time_conn BETWEEN '2023-03-01 00:00:00' AND '2023-03-31 23:59:59'
GROUP  BY cube( Date(time_conn), program_ver )
ORDER  BY Date(time_conn) DESC,
          program_ver DESC;
